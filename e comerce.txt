let cart = [];

/* DISPLAY PREMIUM STYLE */
function displayBooks(data) {
    let list = document.getElementById("bookList");

    list.innerHTML = `
    <div class="row">
        <h2>🔥 Trending Books</h2>
        <div class="row-container">
            ${data.map(book => `
                <div class="book" onclick='addToCart(${JSON.stringify(book)})'>
                    <img src="${book.img}">
                    <div class="info">
                        <h4>${book.name}</h4>
                        <div class="tag">${book.type}</div>
                    </div>
                </div>
            `).join("")}
        </div>
    </div>`;
}

/* CART */
function addToCart(book) {
    cart.push(book);
    updateCart();
}

/* UPDATE CART */
function updateCart() {
    let cartList = document.getElementById("cartItems");
    let total = 0;

    cartList.innerHTML = "";

    cart.forEach((item, i) => {
        total += item.price;
        cartList.innerHTML += `
        <li>
            ${item.name} - ₹${item.price}
            <button onclick="removeItem(${i})">❌</button>
        </li>`;
    });

    document.getElementById("total").innerText = "Total: ₹" + total;
}

/* REMOVE */
function removeItem(index) {
    cart.splice(index, 1);
    updateCart();
}
