<script>
let books = [
    {name:"Atomic Habits",price:499,type:"New",img:"https://m.media-amazon.com/images/I/91bYsX41DVL.jpg"},
    {name:"Rich Dad Poor Dad",price:350,type:"New",img:"https://m.media-amazon.com/images/I/81bsw6fnUiL.jpg"},
    {name:"The Alchemist",price:299,type:"Old",img:"https://m.media-amazon.com/images/I/71aFt4+OTOL.jpg"},
    {name:"Ikigai",price:420,type:"New",img:"https://m.media-amazon.com/images/I/81l3rZK4lnL.jpg"},
    {name:"Think and Grow Rich",price:280,type:"Old",img:"https://m.media-amazon.com/images/I/71UypkUjStL.jpg"},
    {name:"Wings of Fire",price:350,type:"New",img:"https://m.media-amazon.com/images/I/71KKZlVjbwL.jpg"}
];

let cart = [];

/* DISPLAY BOOKS */
function displayBooks(data) {
    let list = document.getElementById("bookList");
    list.innerHTML = "";

    data.forEach((book) => {
        list.innerHTML += `
        <div class="book">
            <img src="${book.img}">
            <h4>${book.name}</h4>
            <p style="color:green;font-weight:bold;">${book.type}</p>
            <p><b>₹${book.price}</b></p>
            <button onclick='addToCart(${JSON.stringify(book)})'>Add to Cart</button>
        </div>`;
    });
}

/* ADD TO CART */
function addToCart(book) {
    cart.push(book);
    alert(book.name + " added to cart 🛒");
    updateCart();
}

/* UPDATE CART */
function updateCart() {
    let cartList = document.getElementById("cartItems");
    let total = 0;
    cartList.innerHTML = "";

    cart.forEach((item, i) => {
        total += item.price;
        cartList.innerHTML += `
        <li>
            ${item.name} - ₹${item.price}
            <button onclick="removeItem(${i})">❌</button>
        </li>`;
    });

    document.getElementById("total").innerText = "Total: ₹" + total;
}

/* REMOVE ITEM */
function removeItem(index) {
    cart.splice(index, 1);
    updateCart();
}

/* SEARCH */
function searchBooks() {
    let input = document.querySelector(".search-box").value.toLowerCase();

    let filtered = books.filter(book =>
        book.name.toLowerCase().includes(input)
    );

    displayBooks(filtered);
}

/* PAYMENT */
function payOnline() {
    alert("Redirecting to payment gateway...");
}

displayBooks(books);
